//experimentando funções da biblioteca Math do JS.

var num1 = 1;
var num2 = 2;
var num3 = 1.5;

//retorna valor absoluto
Math.abs (num3);

//arredonda pro próximo inteiro acima
Math.round (num3);

//arredonda para próximo inteiro abaixo
Math.floor (num3);

//retorna maior número
Math.max (num1, num2);

//retorna menor número
Math.min (num1, num2);
